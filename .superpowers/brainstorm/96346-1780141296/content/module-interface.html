Module Interface Design

This is the contract that each module implements

TypeScript Interface
interface Module {
  id: string;
  name: string;
  description: string;
  keywords: string[];
  requiredInputs: InputField[];
  execute(inputs: Record<string, any>): Promise<ModuleResult>;
}

interface InputField {
  id: string;
  label: string;
  type: 'text' | 'file' | 'select' | 'date';
  required: boolean;
  options?: string[];
  placeholder?: string;
}

interface ModuleResult {
  success: boolean;
  message: string;
  details?: any;
}
